function countDivisible(n: number, k: number): number {

    return Math.floor(n / k)
}

在k 至n 的范围内迭代，步长为k ，并递增一个预定义的计数器。 

function countDivisible(n: number, k: number): number {

    let count = 0;
    for (let i=k; i < n + 1; i+=k) count += 1;
    
    return count;
}
